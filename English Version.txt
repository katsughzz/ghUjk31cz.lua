local Tbl_ActiveNotifications = {}
local Cfg_NotificationSize = Vector2.new(220, 70)
local Cfg_ScreenMargin = Vector2.new(10, 10)
local Cfg_Spacing = 8

local function Func_UpdatePositions()
    local var_currentY = Cfg_ScreenMargin.Y
    for i = 1, #Tbl_ActiveNotifications do
        local obj_frame = Tbl_ActiveNotifications[i]
        if obj_frame and obj_frame.Parent then
            local pos_target = UDim2.new(1, -Cfg_ScreenMargin.X, 1, -var_currentY)
            Srv_Tween:Create(obj_frame, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = pos_target}):Play()
            var_currentY = var_currentY + Cfg_NotificationSize.Y + Cfg_Spacing
        end
    end
end

function Func_Notify(p_title, p_subtitle, p_duration)
    local var_duration = p_duration or 4
    local obj_frame = Instance.new("Frame")
    obj_frame.Size = UDim2.fromOffset(Cfg_NotificationSize.X, Cfg_NotificationSize.Y)
    obj_frame.AnchorPoint = Vector2.new(1, 1)
    obj_frame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
    obj_frame.BorderSizePixel = 0
    obj_frame.Parent = GUI_NotificationHost
    Instance.new("UICorner", obj_frame).CornerRadius = UDim.new(0, 12)
    local obj_title = Instance.new("TextLabel", obj_frame)
    obj_title.Text = p_title or "Notification"
    obj_title.Font = Enum.Font.GothamBold
    obj_title.TextColor3 = Color3.fromRGB(255, 255, 255)
    obj_title.TextSize = 16
    obj_title.BackgroundTransparency = 1
    obj_title.Size = UDim2.new(1, -20, 0, 28)
    obj_title.Position = UDim2.fromOffset(10, 8)
    obj_title.TextXAlignment = Enum.TextXAlignment.Center
    local obj_subtitle = Instance.new("TextLabel", obj_frame)
    obj_subtitle.Text = p_subtitle or "..."
    obj_subtitle.Font = Enum.Font.Gotham
    obj_subtitle.TextColor3 = Color3.fromRGB(200, 200, 200)
    obj_subtitle.TextSize = 14
    obj_subtitle.BackgroundTransparency = 1
    obj_subtitle.Size = UDim2.new(1, -20, 0, 22)
    obj_subtitle.Position = UDim2.fromOffset(10, 38)
    obj_subtitle.TextXAlignment = Enum.TextXAlignment.Center
    local var_startY = Cfg_ScreenMargin.Y
    for i = 1, #Tbl_ActiveNotifications do var_startY = var_startY + Cfg_NotificationSize.Y + Cfg_Spacing end
    local pos_final = UDim2.new(1, -Cfg_ScreenMargin.X, 1, -var_startY)
    obj_frame.Position = pos_final + UDim2.fromOffset(0, 80)
    obj_frame.BackgroundTransparency = 0.5
    Srv_Tween:Create(obj_frame, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {Position = pos_final, BackgroundTransparency = 0}):Play()
    table.insert(Tbl_ActiveNotifications, obj_frame)
    task.delay(var_duration, function()
        if obj_frame and obj_frame.Parent then
            for i, v in ipairs(Tbl_ActiveNotifications) do if v == obj_frame then table.remove(Tbl_ActiveNotifications, i); break; end end
            local anim_out = Srv_Tween:Create(obj_frame, TweenInfo.new(0.3, Enum.EasingStyle.Quad, Enum.EasingDirection.In), {Position = obj_frame.Position + UDim2.fromOffset(0, 80), BackgroundTransparency = 1})
            anim_out:Play()
            Func_UpdatePositions()
            anim_out.Completed:Wait()
            obj_frame:Destroy()
        end
    end)
end

local ID_SoundSuccess = "3625290171"
local ID_SoundError = "2865228021"
local ID_BackgroundImage = "rbxassetid://124288667234232"
local Cfg_ButtonColor = Color3.fromRGB(60, 60, 60)
local Name_PredictableSound = "PainelMusica_UserId_" .. Var_LocalPlayer.UserId
local Name_CurrentSound = nil

local function Util_FindRemoteEvent(p_name)
    for _, service in ipairs({game:GetService("ReplicatedStorage"), game:GetService("Workspace")}) do
        for _, obj in ipairs(service:GetDescendants()) do
            if obj:IsA("RemoteEvent") and obj.Name == p_name then return obj end
        end
    end
    return nil
end

local function Util_PlaySound(p_soundId)
    if p_soundId and tonumber(p_soundId) then
        pcall(function()
            local sound_obj = Instance.new("Sound")
            sound_obj.Name = "NotificationSound"
            sound_obj.SoundId = "rbxassetid://" .. p_soundId
            sound_obj.Volume = 0.7
            sound_obj.Parent = Var_PlayerGui
            sound_obj:Play()
            Srv_Debris:AddItem(sound_obj, sound_obj.TimeLength + 1)
        end)
    end
end

local function Core_PlayMusic(p_musicId, p_volume, p_pitch)
    local remote = Util_FindRemoteEvent("AC6_FE_Sounds")
    if remote then
        remote:FireServer("stopSound", Name_PredictableSound)
        Name_CurrentSound = Name_PredictableSound
        local assetId = "rbxassetid://" .. p_musicId
        remote:FireServer("newSound", Name_CurrentSound, workspace, assetId, p_pitch, p_volume, true)
        remote:FireServer("playSound", Name_CurrentSound)
    else
        warn("AC6_FE_Sounds RemoteEvent not found.")
    end
end

local function Core_StopAllMusic()
    local remote = Util_FindRemoteEvent("AC6_FE_Sounds")
    if not remote then warn("AC6_FE_Sounds RemoteEvent not found."); return; end
    local wasStopped = false
    for _, sound in ipairs(workspace:GetDescendants()) do
        if sound:IsA("Sound") and sound.IsPlaying then
            remote:FireServer("stopSound", sound.Name)
            wasStopped = true
        end
    end
    if wasStopped then Func_Notify("Music", "Stopped.", 3) end
    Name_CurrentSound = nil
end

local function Core_AdoptOrphanedSound()
    local orphaned = workspace:FindFirstChild(Name_PredictableSound)
    if orphaned and orphaned:IsA("Sound") then
        print("Music Panel: Orphaned sound found and adopted.")
        Name_CurrentSound = Name_PredictableSound
    end
end

if not Util_FindRemoteEvent("AC6_FE_Sounds") then
    Util_PlaySound(ID_SoundError)
    Func_Notify("Error", "Game not compatible.", 5)
    return
end

Util_PlaySound(ID_SoundSuccess)
Func_Notify("System", "Music Panel Loaded!", 5)
Core_AdoptOrphanedSound()

Btn_Play.MouseButton1Click:Connect(function()
    local var_musicId = Input_MusicID.Text
    local var_volume = tonumber(Input_Volume.Text) or 1
    local var_pitch = tonumber(Input_Pitch.Text) or 1
    if not tonumber(var_musicId) then Util_PlaySound(ID_SoundError); Func_Notify("Error", "Invalid ID (not a number).", 5); return end
    local success, result = pcall(function() return Srv_Marketplace:GetProductInfo(tonumber(var_musicId)) end)
    if success and result then
        if result.Name == "[ Content Deleted ]" then Util_PlaySound(ID_SoundError); Func_Notify("Error", "This audio has been deleted.", 5); return end
        if result.AssetTypeId == Enum.AssetType.Audio.Value then
            Util_PlaySound(ID_SoundSuccess)
            Func_Notify("Music", "Playing...", 3)
            Core_PlayMusic(var_musicId, var_volume, var_pitch)
        else
            Util_PlaySound(ID_SoundError)
            Func_Notify("Error", "This ID does not belong to an audio.", 5)
        end
    else
        Util_PlaySound(ID_SoundError)
        Func_Notify("Error", "Music ID does not exist.", 5)
    end
end)
